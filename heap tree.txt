#include <stdio.h>
#include <stdlib.h>

#define MAX_ELEMENTS 200
#define HEAP_FULL(n) ((n) == MAX_ELEMENTS - 1)
#define HEAP_EMPTY(n) (!(n))

typedef struct {
	int key;
} element;

element heap[MAX_ELEMENTS];
int hsize = 0;

void push(element item, int *n)
{
	int i;
	if (HEAP_FULL(*n)) {
		fprintf(stderr, "힙이 꽉찼습니다");
		exit(1);
	}

	i = ++(*n);
	while (i != 1 && (item.key > heap[i / 2].key)) {
		heap[i] = heap[i / 2];
		i /= 2;
	}

	heap[i] = item;
}

element pop(int *n)
{
	int parent, child;
	element item, temp;

	if (HEAP_EMPTY(*n)) {
		fprintf(stderr, "힙이 비어있습니다\n");
		exit(1);
	}
	item = heap[1];
	temp = heap[(*n)--];

	parent = 1;
	child = 2;
	while (child <= *n) {
		if (child < *n && heap[child].key < heap[child + 1].key) {
			child++;
		}
		if (temp.key >= heap[child].key) {
			break;
		}
		heap[parent] = heap[child];
		parent = child;
		child *= 2;
	}
	heap[parent] = temp;
	return item;
}

int main()
{
	int i;
	int keys[] = {30, 10, 80, 20, 90, 40, 70, 50, 60};
	int m = (int)(sineof(keys) / sizeof(keys[0]));
	for (i = 0; i < m; i++) {
		element e.key = keys[i];
		push(e, &hsize);
	}

	while (hsize) {
		element e = pop(&hsize);
		printf("%d", e.key);
	}
	printf("\n");
}
