#include <stdlib.h>

typedef struct Node *nodepointer;
typedef struct Node {
	int coef;
	int exp;
	nodepointer next;
} Node;

	a->exp = exp;

void multipoly(nodepointer a, nodepointer b)
{
	nodepointer result = NULL;

	for (nodepointer p = a; p != NULL; p = p->next) {
		for (nodepointer q = b; q != NULL; q = q->next) {
			int c = p->coef * q->coef;
			int e = p->exp + q->exp;

			nodepointer temp = result;
			nodepointer prev = NULL;
			while (temp != NULL && temp->exp > e) {
				prev = temp;
				temp = temp->next;
			}
			if (temp != NULL && temp->exp == e) {
				temp->coef += c;
			} else {
				nodepointer newNode = createNode(c, e);
				if (prev == NULL) {
					newNode->next = result;
					result = newNode;
				} else {
					newNode->next = prev->next;
					prev->next = newNode;
				}
			}
		}
	}
	return result;
}
